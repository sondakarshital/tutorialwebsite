Let's start with first nodeJs Example.

app/firstNodeJsExample.js

var express = require('express');

var app= express();

var bodyParser = require('body-parser')

var cors = require('cors')

//Functions to get and post videos

var getVideos =require('./routes/videos/getVideos.js');

var postVideos =require('./routes/videos/postVideos.js');

//used for CORS Support

app.use(cors());

//Gtes the videos

app.get('/getVideos',getVideos.getVideos);

//posts the videos

app.post('/postVideos', postVideos.postVideos);
